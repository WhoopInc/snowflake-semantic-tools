# SST Environment Variables Template
# Copy this file to .env and fill in your actual values
# 
# IMPORTANT: This file contains sensitive credentials and should NEVER be committed to git
# Ensure the .env file is in your dbt project's .gitignore
#
# CRITICAL: Do NOT use quotes around values - they will be included literally!
# WRONG: SNOWFLAKE_USER='user@company.com'  (includes quotes in value)
# RIGHT: SNOWFLAKE_USER=user@company.com    (correct format)

# ============================================================================
# Snowflake Connection Details (Required)
# ============================================================================
SNOWFLAKE_ACCOUNT=<YOUR_ACCOUNT>
SNOWFLAKE_USER=<YOUR_EMAIL>
# Note: SNOWFLAKE_USERNAME is also supported (for dbt compatibility)
# If both are set, SNOWFLAKE_USER takes precedence
SNOWFLAKE_WAREHOUSE=<YOUR_WAREHOUSE>
SNOWFLAKE_ROLE=<YOUR_ROLE>

# ============================================================================
# Snowflake Authentication (Choose ONE method)

# Method 1: Browser SSO (RECOMMENDED for local development)

#
# No additional configuration needed - just leave the lines below commented out

# Method 2: RSA Key Authentication (RECOMMENDED for CI/CD)
# ─────────────────────────────────────────────────────────
